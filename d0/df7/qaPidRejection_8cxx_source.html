164 
165  TrackSelectorPID selectorMuon(kMuonPlus);
166 
167  std::vector<int64_t> recoEvt(collisions.size());
168  std::vector<int64_t> recoTracks(tracks.size());
169  LOGF(info, "%d", particlePDG);
170  for (const auto& track : tracks) {
171  const auto mcParticle = track.mcParticle();
172  if (particlePDG != 0 && mcParticle.pdgCode() != particlePDG) { // Checking PDG code
173  continue;

175  bool isTOFhpElectron = !(selectorElectron.getStatusTrackPIDTOF(track) == TrackSelectorPID::Status::PIDRejected);
176  bool isRICHhpElectron = !(selectorElectron.getStatusTrackPIDRICH(track) == TrackSelectorPID::Status::PIDRejected);
177  bool isTOFhpPion = !(selectorPion.getStatusTrackPIDTOF(track) == TrackSelectorPID::Status::PIDRejected);
178  bool isRICHhpPion = !(selectorPion.getStatusTrackPIDRICH(track) == TrackSelectorPID::Status::PIDRejected);
179  bool isTOFhpKaon = !(selectorKaon.getStatusTrackPIDTOF(track) == TrackSelectorPID::Status::PIDRejected);
180  bool isRICHhpKaon = !(selectorKaon.getStatusTrackPIDRICH(track) == TrackSelectorPID::Status::PIDRejected);
181  bool isTOFhpProton = !(selectorProton.getStatusTrackPIDTOF(track) == TrackSelectorPID::Status::PIDRejected);
182  bool isRICHhpProton = !(selectorProton.getStatusTrackPIDRICH(track) == TrackSelectorPID::Status::PIDRejected);
183  bool isMIDhpMuon = (selectorMuon.getStatusTrackPIDMID(track) == TrackSelectorPID::Status::PIDAccepted);
184 
185  if (MC::isPhysicalPrimary(mcParticle)) {
186  histos.fill(HIST("tracking/pteta"), track.pt(), track.eta());
187  if (isTOFhpElectron) {
188  histos.fill(HIST("trackingTOFselElectron/pteta"), track.pt(), track.eta());
189  }
190  if (isTOFhpPion) {
191  histos.fill(HIST("trackingTOFselPion/pteta"), track.pt(), track.eta());
192  }
193  if (isTOFhpKaon) {
194  histos.fill(HIST("trackingTOFselKaon/pteta"), track.pt(), track.eta());
195  }
196  if (isTOFhpProton) {
197  histos.fill(HIST("trackingTOFselProton/pteta"), track.pt(), track.eta());
198  }
199  if (isRICHhpElectron) {
200  histos.fill(HIST("trackingRICHselElectron/pteta"), track.pt(), track.eta());
201  }
202  if (isRICHhpPion) {
203  histos.fill(HIST("trackingRICHselPion/pteta"), track.pt(), track.eta());
204  }
205  if (isRICHhpKaon) {
206  histos.fill(HIST("trackingRICHselKaon/pteta"), track.pt(), track.eta());
207  }
208  if (isRICHhpProton) {
209  histos.fill(HIST("trackingRICHselProton/pteta"), track.pt(), track.eta());
210  }
211  if (isMIDhpMuon) {
212  histos.fill(HIST("trackingMIDselMuon/pteta"), track.pt(), track.eta());
213  }
214 
215  histos.fill(HIST("tracking/peta"), track.p(), track.eta());
216  if (isTOFhpElectron) {
217  histos.fill(HIST("trackingTOFselElectron/peta"), track.p(), track.eta());
218  }
219  if (isTOFhpPion) {
220  histos.fill(HIST("trackingTOFselPion/peta"), track.p(), track.eta());
221  }
222  if (isTOFhpKaon) {
223  histos.fill(HIST("trackingTOFselKaon/peta"), track.p(), track.eta());
224  }
225  if (isTOFhpProton) {
226  histos.fill(HIST("trackingTOFselProton/peta"), track.p(), track.eta());
227  }
228  if (isRICHhpElectron) {
229  histos.fill(HIST("trackingRICHselElectron/peta"), track.p(), track.eta());
230  }

242  }
243  }

340  selectorElectron.setRangeNSigmaTPC(-d_nSigmaTPC, d_nSigmaTPC);

359  for (const auto& track : tracks) {
360 
361  if (std::abs(track.eta()) > etaMaxSel || track.pt() < ptMinSel) {
362  continue;
363  }
364  const auto mcParticle = track.mcParticle();
365  histos.fill(HIST("hAllNoSel/pteta"), track.pt(), track.eta());
366  if (mcParticle.pdgCode() == kElectron) {
367  histos.fill(HIST("hElectronNoSel/pteta"), track.pt(), track.eta());
368  }
369  if (mcParticle.pdgCode() == kMuonPlus) {
370  histos.fill(HIST("hMuonNoSel/pteta"), track.pt(), track.eta());
371  }
372  if (mcParticle.pdgCode() == kPiPlus) {
373  histos.fill(HIST("hPionNoSel/pteta"), track.pt(), track.eta());
374  }
