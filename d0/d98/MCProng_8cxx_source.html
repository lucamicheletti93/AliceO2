114 bool MCProng::TestPDG(int i, int pdgCode) const
115 {
116  if (i < 0 || i >= fNGenerations) { // TODO: need an error message
117  return false;

119  if (!ComparePDG(pdgCode, fPDGcodes[i], fCheckBothCharges[i], fExcludePDG[i])) {

126 bool MCProng::ComparePDG(int pdg, int prongPDG, bool checkBothCharges, bool exclude) const
127 {
128  //
129  // test whether the particle pdg ("pdg") matches this MC signal prong ("prongPDG")
130  //
131  bool decision = true;
132  // compute the absolute value of the PDG code
133  int absPDG = std::abs(pdg);
134  switch (std::abs(prongPDG)) {
135  case kPDGCodeNotAssigned:
136  // PDG not required (any code will do fine)
137  break;
138  case 100: // light flavoured mesons
139  if (checkBothCharges) {
140  decision = absPDG >= 100 && absPDG <= 199;
141  } else {
142  decision = (prongPDG > 0 ? pdg >= 100 && pdg <= 199 : pdg >= -199 && pdg <= -100);
143  }
144  break;
145  case 1000: // light flavoured baryons
146  if (checkBothCharges) {
147  decision = absPDG >= 1000 && absPDG <= 1999;
148  } else {
149  decision = (prongPDG > 0 ? pdg >= 1000 && pdg <= 1999 : pdg >= -1999 && pdg <= -1000);
150  }
151  break;
152  case 200: // light flavoured mesons
153  if (checkBothCharges) {
154  decision = absPDG >= 200 && absPDG <= 299;
155  } else {
156  decision = (prongPDG > 0 ? pdg >= 200 && pdg <= 299 : pdg >= -299 && pdg <= -200);
157  }
158  break;
159  case 2000: // light flavoured baryons
160  if (checkBothCharges) {
161  decision = absPDG >= 2000 && absPDG <= 2999;
162  } else {
163  decision = (prongPDG > 0 ? pdg >= 2000 && pdg <= 2999 : pdg >= -2999 && pdg <= -2000);
164  }
165  break;
166  case 300: // all strange mesons
167  if (checkBothCharges) {
168  decision = absPDG >= 300 && absPDG <= 399;
169  } else {
170  decision = (prongPDG > 0 ? pdg >= 300 && pdg <= 399 : pdg >= -399 && pdg <= -300);
171  }
172  break;
173  case 3000: // all strange baryons
174  if (checkBothCharges) {
175  decision = absPDG >= 3000 && absPDG <= 3999;
176  } else {
177  decision = (prongPDG > 0 ? pdg >= 3000 && pdg <= 3999 : pdg >= -3999 && pdg <= -3000);
178  }
179  break;
180  case 400: // all charmed mesons
181  if (checkBothCharges) {
182  decision = absPDG >= 400 && absPDG <= 499;
183  } else {
184  decision = (prongPDG > 0 ? pdg >= 400 && pdg <= 499 : pdg >= -499 && pdg <= -400);
185  }
186  break;
187  case 401: // open charm mesons
188  if (checkBothCharges) {
189  decision = absPDG >= 400 && absPDG <= 439;
190  } else {
191  decision = (prongPDG > 0 ? pdg >= 400 && pdg <= 439 : pdg >= -439 && pdg <= -400);
192  }
193  break;
194  case 402: // open charm mesons and baryons together
195  if (checkBothCharges) {
196  decision = (absPDG >= 400 && absPDG <= 439) || (absPDG >= 4000 && absPDG <= 4399);
197  } else {
198  if (prongPDG > 0) {
199  decision = (pdg >= 400 && pdg <= 439) || (pdg >= 4000 && pdg <= 4399);
200  }
201  if (prongPDG < 0) {
202  decision = (pdg >= -439 && pdg <= -400) || (pdg >= -4399 && pdg <= -4000);
203  }
204  }
205  break;
206  case 403: // all charm hadrons
207  if (checkBothCharges) {
208  decision = (absPDG >= 400 && absPDG <= 499) || (absPDG >= 4000 && absPDG <= 4999);
209  } else {
210  if (prongPDG > 0) {
211  decision = (pdg >= 400 && pdg <= 499) || (pdg >= 4000 && pdg <= 4999);
212  }
213  if (prongPDG < 0) {
214  decision = (pdg >= -499 && pdg <= -400) || (pdg >= -4999 && pdg <= -4000);
215  }
216  }
217  break;
218  case 404: // charged open charmed mesons w/o s-quark
219  if (checkBothCharges) {
220  decision = (absPDG >= 410 && absPDG <= 419);
221  } else {
222  decision = (prongPDG > 0 ? pdg >= 410 && pdg <= 419 : pdg >= -419 && pdg <= -410);
223  }
224  break;
225  case 405: // neutral open charmed mesons
226  if (checkBothCharges) {
227  decision = absPDG >= 420 && absPDG <= 429;
228  } else {
229  decision = (prongPDG > 0 ? pdg >= 420 && pdg <= 429 : pdg >= -429 && pdg <= -420);
230  }
231  break;
232  case 406: // charged open charmed mesons with s-quark

234  decision = (absPDG >= 430 && absPDG <= 439);
235  } else {
236  decision = (prongPDG > 0 ? pdg >= 430 && pdg <= 439 : pdg >= -439 && pdg <= -430);
237  }

324  }
325  break;
326  case 902: // open charm,beauty mesons and baryons together
327  if (checkBothCharges) {
328  decision = (absPDG >= 400 && absPDG <= 439) || (absPDG >= 4000 && absPDG <= 4399) ||
329  (absPDG >= 500 && absPDG <= 549) || (absPDG >= 5000 && absPDG <= 5499);
330  } else {
331  if (prongPDG > 0) {
332  decision = (pdg >= 400 && pdg <= 439) || (pdg >= 4000 && pdg <= 4399) ||
333  (pdg >= 500 && pdg <= 549) || (pdg >= 5000 && pdg <= 5499);
334  }
335  if (prongPDG < 0) {
336  decision = (pdg >= -439 && pdg <= -400) || (pdg >= -4399 && pdg <= -4000) ||
337  (pdg >= -549 && pdg <= -500) || (pdg >= -5499 && pdg <= -5000);
338  }
339  }
340  break;
341  case 903: // all hadrons in the code range 100-599, 1000-5999
342  if (checkBothCharges) {
343  decision = (absPDG >= 100 && absPDG <= 599) || (absPDG >= 1000 && absPDG <= 5999);
344  } else {
345  if (prongPDG > 0) {
346  decision = (pdg >= 100 && pdg <= 599) || (pdg >= 1000 && pdg <= 5999);

349  decision = (pdg >= -599 && pdg <= -100) || (pdg >= -5999 && pdg <= -1000);
350  }
351  }
352  break;
353  default: // all explicit PDG code cases
354  if (checkBothCharges) {
355  decision = (std::abs(prongPDG) == absPDG);
356  } else {
357  decision = (prongPDG == pdg);
358  }
359  }
360 
361  // if exclude is specified, then reverse decision
362  if (prongPDG && exclude) {
363  decision = !decision;
364  }
365  return decision;
