213 
214  int i = 0;

224  }
225

317  void process(soa::Filtered<MyEventsVtxCovSelected>::iterator const& event, soa::Filtered<MyBarrelTracksSelected> const& tracks, ReducedMCTracks const& tracksMC)
318  {
319  // Reset the fValues array
320  VarManager::ResetValues(0, VarManager::kNVars, fValues);
321  VarManager::FillEvent<gkEventFillMap>(event, fValues);
322 
323  // Run the same event pairing for barrel tracks
324  uint8_t twoTrackFilter = 0;
325  uint16_t dileptonFilterMap = 0;
326  // Loop over reconstructed pairs and fill histograms
327  for (auto& [t1, t2] : combinations(tracks, tracks)) {
328  twoTrackFilter = t1.isBarrelSelected() & t2.isBarrelSelected();
329  if (!twoTrackFilter) { // the tracks must have at least one filter bit in common to continue
330  continue;
331  }
332  VarManager::FillPair(t1, t2, fValues);
333  VarManager::FillPairVertexing(event, t1, t2, fValues);
334  for (int i = 0; i < fCutNames.size(); ++i) {
335  if (twoTrackFilter & (uint8_t(1) << i)) {
336  if (t1.sign() * t2.sign() < 0) {
337  fHistMan->FillHistClass(Form("PairsBarrelSEPM_%s", fCutNames[i].Data()), fValues);
338  for (auto& sig : fRecMCSignals) {
339  if (sig.CheckSignal(false, tracksMC, t1.reducedMCTrack(), t2.reducedMCTrack())) {
340  fHistMan->FillHistClass(Form("PairsBarrelSEPM_%s_%s", fCutNames[i].Data(), sig.GetName()), fValues);
341  }
342  }
343  } else {
344  if (t1.sign() > 0) {
345  fHistMan->FillHistClass(Form("PairsBarrelSEPP_%s", fCutNames[i].Data()), fValues);
346  } else {
347  fHistMan->FillHistClass(Form("PairsBarrelSEMM_%s", fCutNames[i].Data()), fValues);
348  }
349  }
350  }
351  }
352  } // end loop over barrel track pairs
353 
354  // loop over mc stack and fill histograms for pure MC truth signals
355  for (auto& mctrack : tracksMC) {
356  VarManager::FillTrack<gkParticleMCFillMap>(mctrack, fValues);
357  // NOTE: Signals are checked here based on the skimmed MC stack, so depending on the requested signal, the stack could be incomplete.
358  // NOTE: However, the working model is that the decisions on MC signals are precomputed during skimming and are stored in the mcReducedFlags member.
359  // TODO: Use the mcReducedFlags to select signals
360  for (auto& sig : fGenMCSignals) {
361  if (sig.CheckSignal(false, tracksMC, mctrack)) {
362  fHistMan->FillHistClass(Form("MCTruthGen_%s", sig.GetName()), fValues);
363  }
364  }
365  }
366  }
