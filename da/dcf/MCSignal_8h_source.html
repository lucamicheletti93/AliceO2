14 /* Schematic Monte Carlo signal definition:
15 
16  Prong_0: Particle(0,0) <-- Particle(0,1) <-- ... Prong(0,m_0) ... <-- Particle(0, n_0)
17  |
18  Prong_1: Particle(1,0) <-- Particle(1,1) <-- ... Prong(1,m_1) ... <-- Particle(1, n_1)
19  |
20  Prong_i: Particle(i,0) <-- Particle(i,1) <-- ... Prong(i,m_i) ... <-- Particle(i, n_i)
21  |
22  Prong_N: Particle(N,0) <-- Particle(N,1) <-- ... Prong(N,m_N) ... <-- Particle(N, n_N)
23 
24 The MC signal model is composed of N prongs (see MCProng.h) with a variable number of generations specified for each prong (n_i, i=1,N).
25 At least one prong has to be specified for a valid MCsignal. An optional common ancestor for all prongs can be specified (m_i, i=1,N).
26 The common ancestor index, m_i, does not need to be the same for all prongs (e.g. some prongs can have a smaller number of generations in the history up to the common ancestor).
27 When a tuple of MC particles is checked whether it matches the specified signal, all prongs must be matched exactly with the particles in the tuple.
28 Example usage:
29 
30 ATask {
31 
32 MCSignal* mySignal;
33 MCSignal* mySignal2;
34 
35 init() {
36  MCProng prongElectronNonPromptJpsi(3,{11,443,502},{true,true,true},{false,false,false},{0,0,0},{0,0,0},{false,false,false});
37  mySignal = new MCSignal("jpsiBeautyElectron", "Electrons from beauty jpsi decays", {prongElectronNonPromptJpsi}, {-1});
38  MCProng prongAllFromBeauty(2,{0,503},{true,true},{false,false},{0,0},{0,0},{false,false});
39  mySignal2 = new MCSignal("everythingFromBeautyPairs", "Everything from beauty hadrons pair", {prongAllFromBeauty,prongAllFromBeauty}, {-1,-1});
40 }
41 process(aod::McParticles const& mcTracks) {
42  ...
43  for (auto& mctrack : mcTracks) {
44  if(mySignal.CheckSignal(true,mcTracks,mctrack)) {
45  cout << "Found signal" << endl;
46  }
47  }
48  for (auto& [mt1, mt2] : combinations(mcTracks, mcTracks)) {
49  if(mySignal2.CheckSignal(true,mcTracks,mt1,mt2)) {
50  cout << "Match found for " << mySignal2.GetName() << endl;
51  }
52  }
53 }
54 }

131 template <typename U, typename T>
132 bool MCSignal::CheckProng(int i, bool checkSources, const U& mcStack, const T& track)
133 {
134  auto currentMCParticle = track;
135  // loop over the generations specified for this prong
136  for (int j = 0; j < fProngs[i].fNGenerations; j++) {
137  // check the PDG code
138  if (!fProngs[i].TestPDG(j, currentMCParticle.pdgCode())) {
139  return false;
140  }
141  // check the common ancestor (if specified)
142  if (fNProngs > 1 && fCommonAncestorIdxs[i] == j) {
143  if (i == 0) {
144  fTempAncestorLabel = currentMCParticle.globalIndex();
145  } else {
146  if (currentMCParticle.globalIndex() != fTempAncestorLabel) {
147  return false;
148  }
149  }
150  }
151  // make sure that a mother exists in the stack before moving one generation further in history
152  if (!currentMCParticle.has_mother0() && j < fProngs[i].fNGenerations - 1) {
153  return false;
154  }
155  if (j < fProngs[i].fNGenerations - 1) {
156  currentMCParticle = mcStack.iteratorAt(currentMCParticle.mother0Id());
157  //currentMCParticle = currentMCParticle.template mother0_as<U>();
158  }
159  }
160  // NOTE: Checking the sources is optional due to the fact that with the skimmed data model
161  // the full history of a particle is not guaranteed to be present in the particle stack,
162  // which means that some sources cannot be properly checked (e.g. ALICE physical primary)
163  // However, in the skimmed data model, this kind of checks should had been already computed at skimming time
164  // and present in the decision bit map.
165  if (checkSources) {
166  currentMCParticle = track;
167  for (int j = 0; j < fProngs[i].fNGenerations; j++) {
168  if (!fProngs[i].fSourceBits[j]) {
169  // no sources required for this generation
170  continue;
171  }
172  // check each source
173  uint64_t sourcesDecision = 0;
174  // Check kPhysicalPrimary
175  //TODO: compilation fails if MC.h is included (error: multiple definition of MC::isStable()). Check it out!
176  /*if(fProngs[i].fSourceBits[j] & MCProng::kPhysicalPrimary) {
177  cout << "request kPhysicalPrimary, pdg " << currentMCParticle.pdgCode() << endl;
178  if((fProngs[i].fExcludeSource[j] & MCProng::kPhysicalPrimary) != MC::isPhysicalPrimary<U>(track)) {
179  sourcesDecision |= MCProng::kPhysicalPrimary;
180  cout << "good " << sourcesDecision << endl;
181  }
182  }*/
183  // Check kProducedInTransport
184  if (fProngs[i].fSourceBits[j] & MCProng::kProducedInTransport) {
185  cout << "request kProducedInTransport, pdg " << currentMCParticle.pdgCode() << endl;
186  cout << "1::" << (fProngs[i].fExcludeSource[j] & MCProng::kProducedInTransport) << endl;
187  cout << "2::" << (currentMCParticle.flags() & (uint8_t(1) << 0)) << endl;
188  if ((fProngs[i].fExcludeSource[j] & MCProng::kProducedInTransport) != (currentMCParticle.flags() & (uint8_t(1) << 0))) {
189  sourcesDecision |= MCProng::kProducedInTransport;
190  cout << "good " << sourcesDecision << endl;
191  }
192  }
193  // no source bit is fulfilled
194  if (!sourcesDecision) {
195  return false;
196  }
197  // if fUseANDonSourceBitMap is on, request all bits
198  if (fProngs[i].fUseANDonSourceBitMap[j] && (sourcesDecision != fProngs[i].fSourceBits[j])) {
199  return false;
200  }
201  // move one generation back in history
202  if (j < fProngs[i].fNGenerations - 1) {
203  currentMCParticle = mcStack.iteratorAt(currentMCParticle.mother0Id());
204  //currentMCParticle = currentMCParticle.template mother0_as<U>();
205  }
206  }
207  }
208 
209  return true;
210 }
