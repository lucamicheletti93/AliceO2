41 template <typename T>
42 std::tuple<std::vector<std::size_t>, std::vector<T>, std::vector<T>> DuplicateJetsAroundPhiBoundary(
43  std::vector<T>& jetsPhi,
44  std::vector<T>& jetsEta,
45  double maxMatchingDistance,
46  // TODO: Remove additional margin after additional testing.
47  double additionalMargin = 0.05)
48 {
49  const std::size_t nJets = jetsPhi.size();
50  std::vector<std::size_t> jetsMapToJetIndex(nJets);
51  // We need to keep track of the map from the duplicated vector to the existing jets.
52  // To start, we fill this map (practically, it maps from vector index to an index, so we
53  // just use a standard vector) with the existing jet indices, which range from 0..nJets.
54  std::iota(jetsMapToJetIndex.begin(), jetsMapToJetIndex.end(), 0);
55 
56  // The full duplicated jets will be stored in a new vector to avoid disturbing the input data.
57  std::vector<T> jetsPhiComparison(jetsPhi);
58  std::vector<T> jetsEtaComparison(jetsEta);
59 
60  // Duplicate the jets
61  // When a jet is outside of the desired phi range, we make a copy of the jet, shifting it by 2pi
62  // as necessary. The eta value is copied directly, as is the index of the original jet, so we can
63  // map from index in duplicated data -> index in original collection.
64  // NOTE: Assumes, but does not validate, that 0 <= phi < 2pi.
65  for (std::size_t i = 0; i < nJets; i++) {
66  // Handle lower edge
67  if (jetsPhi[i] <= (maxMatchingDistance + additionalMargin)) {
68  jetsPhiComparison.emplace_back(jetsPhi[i] + 2 * M_PI);
69  jetsEtaComparison.emplace_back(jetsEta[i]);
70  jetsMapToJetIndex.emplace_back(jetsMapToJetIndex[i]);
71  }
72  // Handle upper edge
73  if (jetsPhi[i] >= (2 * M_PI - (maxMatchingDistance + additionalMargin))) {
74  jetsPhiComparison.emplace_back(jetsPhi[i] - 2 * M_PI);
75  jetsEtaComparison.emplace_back(jetsEta[i]);
76  jetsMapToJetIndex.emplace_back(jetsMapToJetIndex[i]);
77  }
78  }
79 
80  return {jetsMapToJetIndex, jetsPhiComparison, jetsEtaComparison};
81 }

106 template <typename T>
107 std::tuple<std::vector<int>, std::vector<int>> MatchJetsGeometricallyImpl(
108  const std::vector<T>& jetsBasePhi,
109  const std::vector<T>& jetsBaseEta,
110  std::vector<T> jetsBasePhiForMatching,
111  std::vector<T> jetsBaseEtaForMatching,
112  const std::vector<std::size_t> jetMapBaseToJetIndex,
113  const std::vector<T>& jetsTagPhi,
114  const std::vector<T>& jetsTagEta,
115  std::vector<T> jetsTagPhiForMatching,
116  std::vector<T> jetsTagEtaForMatching,
117  const std::vector<std::size_t> jetMapTagToJetIndex,
118  double maxMatchingDistance)
119 {
120  // Validation
121  // If no jets in either collection, then return immediately.
122  const std::size_t nJetsBase = jetsBaseEta.size();
123  const std::size_t nJetsTag = jetsTagEta.size();
124  if (!(nJetsBase && nJetsTag)) {
125  return std::make_tuple(std::vector<int>(nJetsBase), std::vector<int>(nJetsTag));
126  }
127  // Require that the comparison vectors are greater than or equal to the standard collections.
128  if (jetsBasePhiForMatching.size() < jetsBasePhi.size()) {
129  throw std::invalid_argument("Base collection phi for matching is smaller than the input base collection.");
130  }
131  if (jetsBaseEtaForMatching.size() < jetsBaseEta.size()) {
132  throw std::invalid_argument("Base collection eta for matching is smaller than the input base collection.");
133  }
134  if (jetsTagPhiForMatching.size() < jetsTagPhi.size()) {
135  throw std::invalid_argument("Tag collection phi for matching is smaller than the input tag collection.");
136  }
137  if (jetsTagEtaForMatching.size() < jetsTagEta.size()) {
138  throw std::invalid_argument("Tag collection eta for matching is smaller than the input tag collection.");
139  }
140 
141  // Build the KD-trees using vectors
142  // We build two trees:
143  // treeBase, which contains the base collection.
144  // treeTag, which contains the tag collection.
145  // The trees are built to match in two dimensions (eta, phi)
146  TKDTree<int, T> treeBase(jetsBaseEtaForMatching.size(), 2, 1), treeTag(jetsTagEtaForMatching.size(), 2, 1);
147  // By utilizing SetData, we can avoid having to copy the data again.
148  treeBase.SetData(0, jetsBaseEtaForMatching.data());
149  treeBase.SetData(1, jetsBasePhiForMatching.data());
150  treeBase.Build();
151  treeTag.SetData(0, jetsTagEtaForMatching.data());
152  treeTag.SetData(1, jetsTagPhiForMatching.data());
153  treeTag.Build();
154 
155  // Storage for the jet matching indices.
156  // matchIndexTag maps from the base index to the tag index.
157  // matchBaseTag maps from the tag index to the base index.
158  std::vector<int> matchIndexTag(nJetsBase, -1), matchIndexBase(nJetsTag, -1);
159 
160  // Find the tag jet closest to each base jet.
161  for (std::size_t iBase = 0; iBase < nJetsBase; iBase++) {
162  Double_t point[2] = {jetsBaseEta[iBase], jetsBasePhi[iBase]};
163  Int_t index(-1);
164  Double_t distance(-1);
165  treeTag.FindNearestNeighbors(point, 1, &index, &distance);
166  // test whether indices are matching:
167  if (index >= 0 && distance < maxMatchingDistance) {
168  LOG(DEBUG) << "Found closest tag jet for " << iBase << " with match index " << index << " and distance " << distance << "\n";
169  matchIndexTag[iBase] = index;
170  } else {
171  LOG(DEBUG) << "Closest tag jet not found for " << iBase << ", distance to closest " << distance << "\n";
172  }
173  }
174 
175  // Find the base jet closest to each tag jet
176  for (std::size_t iTag = 0; iTag < nJetsTag; iTag++) {
177  Double_t point[2] = {jetsTagEta[iTag], jetsTagPhi[iTag]};
178  Int_t index(-1);
179  Double_t distance(-1);
180  treeBase.FindNearestNeighbors(point, 1, &index, &distance);
181  if (index >= 0 && distance < maxMatchingDistance) {
182  LOG(DEBUG) << "Found closest base jet for " << iTag << " with match index " << index << " and distance " << distance << std::endl;
183  matchIndexBase[iTag] = index;
184  } else {
185  LOG(DEBUG) << "Closest tag jet not found for " << iTag << ", distance to closest " << distance << "\n";
186  }
187  }
188 
189  // Convert indices in the duplicated jet vectors into the original jet indices.
190  // First for the base -> tag map.
191  for (auto& v : matchIndexTag) {
192  // If it's -1, it means that it didn't find a matching jet.
193  // We have to explicitly check for it here because it would be an invalid index.
194  if (v != -1) {
195  v = jetMapTagToJetIndex[v];
196  }
197  }
198  // Then for the index -> base map.
199  for (auto& v : matchIndexBase) {
200  if (v != -1) {
201  v = jetMapBaseToJetIndex[v];
202  }
203  }
204 
205  // Finally, we'll check for true matches, which are pairs where the base jet is the
206  // closest to the tag jet and vice versa
207  // As the lists are linear a loop over the outer base jet is sufficient.
208  std::vector<int> baseToTagMap(nJetsBase, -1);
209  std::vector<int> tagToBaseMap(nJetsTag, -1);
210  LOG(DEBUG) << "Starting true jet loop: nbase(" << nJetsBase << "), ntag(" << nJetsTag << ")\n";
211  for (std::size_t iBase = 0; iBase < nJetsBase; iBase++) {
212  LOG(DEBUG) << "base jet " << iBase << ": match index in tag jet container " << matchIndexTag[iBase] << "\n";
213  if (matchIndexTag[iBase] > -1) {
214  LOG(DEBUG) << "tag jet " << matchIndexTag[iBase] << ": matched base jet " << matchIndexBase[matchIndexTag[iBase]] << "\n";
215  }
216  if (matchIndexTag[iBase] > -1 && matchIndexBase[matchIndexTag[iBase]] == iBase) {
217  LOG(DEBUG) << "True match! base index: " << iBase << ", tag index: " << matchIndexTag[iBase] << "\n";
218  baseToTagMap[iBase] = matchIndexTag[iBase];
219  tagToBaseMap[matchIndexTag[iBase]] = iBase;
220  }
221  }
222 
223  return std::make_tuple(baseToTagMap, tagToBaseMap);
224 }

243 template <typename T>
244 std::tuple<std::vector<int>, std::vector<int>> MatchJetsGeometrically(
245  std::vector<T> jetsBasePhi,
246  std::vector<T> jetsBaseEta,
247  std::vector<T> jetsTagPhi,
248  std::vector<T> jetsTagEta,
249  double maxMatchingDistance)
250 {
251  // Validation
252  const std::size_t nJetsBase = jetsBaseEta.size();
253  const std::size_t nJetsTag = jetsTagEta.size();
254  if (!(nJetsBase && nJetsTag)) {
255  // There are no jets, so nothing to be done.
256  return std::make_tuple(std::vector<int>(nJetsBase), std::vector<int>(nJetsTag));
257  }
258  // Input sizes must match
259  if (jetsBasePhi.size() != jetsBaseEta.size()) {
260  throw std::invalid_argument("Base collection eta and phi sizes don't match. Check the inputs.");
261  }
262  if (jetsTagPhi.size() != jetsTagEta.size()) {
263  throw std::invalid_argument("Tag collection eta and phi sizes don't match. Check the inputs.");
264  }
265 
266  // To perform matching with periodic boundary conditions (ie. phi) with a KDTree, we need
267  // to duplicate data up to maxMatchingDistance in phi because phi is periodic.
268  // NOTE: vectors are modified in place to avoid copies.
269  auto&& [jetMapBaseToJetIndex, jetsBasePhiComparison, jetsBaseEtaComparison] = DuplicateJetsAroundPhiBoundary(jetsBasePhi, jetsBaseEta, maxMatchingDistance);
270  auto&& [jetMapTagToJetIndex, jetsTagPhiComparison, jetsTagEtaComparison] = DuplicateJetsAroundPhiBoundary(jetsTagPhi, jetsTagEta, maxMatchingDistance);
271 
272  // Finally, perform the actual matching.
273  auto&& [baseToTagMap, tagToBaseMap] = MatchJetsGeometricallyImpl(
274  jetsBasePhi, jetsBaseEta, jetsBasePhiComparison, jetsBaseEtaComparison, jetMapBaseToJetIndex,
275  jetsTagPhi, jetsTagEta, jetsTagPhiComparison, jetsTagEtaComparison, jetMapTagToJetIndex,
276  maxMatchingDistance);
277 
278  return std::make_tuple(baseToTagMap, tagToBaseMap);
279 }
