14 // Analysis task for skimming MC AODs
15 // Similar to tableMaker.cxx. The written skimmed data model includes, besides the reconstructed data tables, a skimmed MC stack.
16 // The skimmed MC stack includes the MC truth particles corresponding to the list of user specified MC signals (see MCsignal.h)
17 // and the MC truth particles corresponding to the reconstructed tracks selected by the specified track cuts on reconstructed data.
18 // The event MC truth is written as a joinable table with the ReducedEvents table (!simplification wrt Framework data model).
19 // TODO: There are no MC labels for MUON tracks yet

168  void processRec(MyEvents::iterator const& collision, aod::BCs const& bcs, soa::Filtered<MyBarrelTracks> const& tracksBarrel,
169  aod::McParticles const& mcTracks, aod::McCollisions const& mcEvents)
170  {
171  uint32_t triggerAliases = 0;
172  for (int i = 0; i < kNaliases; i++) {
173  if (collision.alias()[i] > 0) {
174  triggerAliases |= (uint32_t(1) << i);
175  }
176  }
177  uint64_t tag = 0; // TODO: add here available computed event cuts (e.g. run2bcinfo().eventCuts()) or other event wise decisions
178 
179  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
180  VarManager::FillEvent<gkEventFillMap>(collision, fValues); // extract event information and place it in the fValues array
181  VarManager::FillEvent<gkEventMCFillMap>(collision.mcCollision(), fValues);
182 
183  fHistMan->FillHistClass("Event_BeforeCuts", fValues); // automatically fill all the histograms in the class Event
184  if (!fEventCut->IsSelected(fValues)) {
185  return;
186  }
187  fHistMan->FillHistClass("Event_AfterCuts", fValues);
188 
189  event(tag, collision.bc().runNumber(), collision.posX(), collision.posY(), collision.posZ(), collision.numContrib());
190  eventExtended(collision.bc().globalBC(), collision.bc().triggerMask(), 0, triggerAliases, fValues[VarManager::kCentVZERO]);
191  eventVtxCov(collision.covXX(), collision.covXY(), collision.covXZ(), collision.covYY(), collision.covYZ(), collision.covZZ(), collision.chi2());
192  eventMC(collision.mcCollision().generatorsID(), collision.mcCollision().posX(), collision.mcCollision().posY(),
193  collision.mcCollision().posZ(), collision.mcCollision().t(), collision.mcCollision().weight(), collision.mcCollision().impactParameter());
194 
195  // loop over the MC truth tracks and find those that need to be written
196  uint16_t mcflags = 0;
197  for (auto& mctrack : mcTracks) {
198  // grouping will be needed here
199  if (mctrack.mcCollision().globalIndex() != collision.mcCollision().globalIndex()) {
200  continue;
201  }
202 
203  // check all the requested MC signals and fill a decision bit map
204  mcflags = 0;
205  int i = 0;
206  for (auto& sig : fMCSignals) {
207  if (sig.CheckSignal(true, mcTracks, mctrack)) {
208  mcflags |= (uint16_t(1) << i);
209  }
210  i++;
211  }
212  // if any of the MC signals was matched, then fill histograms and write that MC particle into the new stack
213  if (mcflags) {
214  // fill histograms for each of the signals, if found
215  VarManager::FillTrack<gkParticleMCFillMap>(mctrack, fValues);
216  for (int i = 0; i < fMCSignals.size(); i++) {
217  if (mcflags & (uint16_t(1) << i)) {
218  fHistMan->FillHistClass(Form("MCtruth_%s", fMCSignals[i].GetName()), fValues);
219  }
220  }
221 
222  fNewLabels[mctrack.index()] = fCounter;
223  fNewLabelsReversed[fCounter] = mctrack.index();
224  fMCFlags[mctrack.index()] = mcflags;
225  fEventIdx[mctrack.index()] = event.lastIndex();
226  fCounter++;
227  }
228  } // end loop over mc stack
229 
230  // loop over reconstructed tracks
231  uint64_t trackFilteringTag = 0;
232  //uint16_t mcflags = 0;
233  trackBasic.reserve(tracksBarrel.size());
234  trackBarrel.reserve(tracksBarrel.size());
235  trackBarrelCov.reserve(tracksBarrel.size());
236  trackBarrelPID.reserve(tracksBarrel.size());
237  trackBarrelLabels.reserve(tracksBarrel.size());
238  for (auto& track : tracksBarrel) {
239  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
240 
241  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
242  if (!fTrackCut->IsSelected(fValues)) {
243  continue;
244  }
245  auto mctrack = track.mcParticle();
246  VarManager::FillTrack<gkParticleMCFillMap>(mctrack, fValues);
247  fHistMan->FillHistClass("TrackBarrel_AfterCuts", fValues);
248 
249  // if the MC truth particle corresponding to this reconstructed track is not already written,
250  // add it to the skimmed stack
251  mcflags = 0;
252  if (!(fNewLabels.find(mctrack.index()) != fNewLabels.end())) {
253  int i = 0;
254  // check all the specified signals
255  for (auto& sig : fMCSignals) {
256  if (sig.CheckSignal(true, mcTracks, mctrack)) {
257  mcflags |= (uint16_t(1) << i);
258  fHistMan->FillHistClass(Form("TrackBarrel_AfterCuts_%s", sig.GetName()), fValues); // fill the reconstructed truth
259  fHistMan->FillHistClass(Form("MCtruth_%s", sig.GetName()), fValues); // fill the generated truth
260  }
261  i++;
262  }
263 
264  fNewLabels[mctrack.index()] = fCounter;
265  fNewLabelsReversed[fCounter] = mctrack.index();
266  fMCFlags[mctrack.index()] = mcflags;
267  fEventIdx[mctrack.index()] = event.lastIndex();
268  fCounter++;
269  }
270 
271  if (track.isGlobalTrack()) {
272  trackFilteringTag |= (uint64_t(1) << 0);
273  }
274  if (track.isGlobalTrackSDD()) {
275  trackFilteringTag |= (uint64_t(1) << 1);
276  }
277  trackBasic(event.lastIndex(), track.globalIndex(), trackFilteringTag, track.pt(), track.eta(), track.phi(), track.sign());
278  trackBarrel(track.tpcInnerParam(), track.flags(), track.itsClusterMap(), track.itsChi2NCl(),
279  track.tpcNClsFindable(), track.tpcNClsFindableMinusFound(), track.tpcNClsFindableMinusCrossedRows(),
280  track.tpcNClsShared(), track.tpcChi2NCl(),
281  track.trdChi2(), track.trdPattern(), track.tofChi2(),
282  track.length(), track.dcaXY(), track.dcaZ());
283  trackBarrelCov(track.x(), track.alpha(), track.y(), track.z(), track.snp(), track.tgl(), track.signed1Pt(),
284  track.cYY(), track.cZY(), track.cZZ(), track.cSnpY(), track.cSnpZ(),
285  track.cSnpSnp(), track.cTglY(), track.cTglZ(), track.cTglSnp(), track.cTglTgl(),
286  track.c1PtY(), track.c1PtZ(), track.c1PtSnp(), track.c1PtTgl(), track.c1Pt21Pt2());
287  trackBarrelPID(track.tpcSignal(),
288  track.tpcNSigmaEl(), track.tpcNSigmaMu(),
289  track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(),
290  track.beta(),
291  track.tofNSigmaEl(), track.tofNSigmaMu(),
292  track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(),
293  track.trdSignal());
294  trackBarrelLabels(fNewLabels.find(mctrack.index())->second, track.mcMask(), mcflags);
295  } // end loop over reconstructed tracks
296  }

298  void processWriteStack(aod::McParticles const& mcTracks)
299  {
300  // loop over the selected MC tracks and write them to the new stack
301  // NOTE: At this point, all the particles which should go into the new stack should had been
302  // selected, such that the mother/daughter new labels can be computed
303  // NOTE: We need to write the particles to the new stack in the exact same order as the
304  // one in which the new labels were assigned. So we use the fNewLabelsReversed which has as key the new label
305  // and as value the old stack label, which we then use to retrieve the particle to be written
306  for (const auto& [newLabel, oldLabel] : fNewLabelsReversed) {
307  auto mctrack = mcTracks.iteratorAt(oldLabel);
308  int mcflags = fMCFlags.find(oldLabel)->second;
309 
310  int m0Label = -1;
311  if (mctrack.has_mother0() && (fNewLabels.find(mctrack.mother0Id()) != fNewLabels.end())) {
312  m0Label = fNewLabels.find(mctrack.mother0Id())->second;
313  }
314  int m1Label = -1;
315  if (mctrack.has_mother1() && (fNewLabels.find(mctrack.mother1Id()) != fNewLabels.end())) {
316  m1Label = fNewLabels.find(mctrack.mother1Id())->second;
317  }
318  int d0Label = -1;
319  if (mctrack.has_daughter0() && (fNewLabels.find(mctrack.daughter0Id()) != fNewLabels.end())) {
320  d0Label = fNewLabels.find(mctrack.daughter0Id())->second;
321  }
322  int d1Label = -1;
323  if (mctrack.has_daughter1() && (fNewLabels.find(mctrack.daughter1Id()) != fNewLabels.end())) {
324  d1Label = fNewLabels.find(mctrack.daughter1Id())->second;
325  }
326  trackMC(fEventIdx.find(oldLabel)->second, mctrack.pdgCode(), mctrack.statusCode(), mctrack.flags(),
327  m0Label, m1Label, d0Label, d1Label,
328  mctrack.weight(), mctrack.pt(), mctrack.eta(), mctrack.phi(), mctrack.e(),
329  mctrack.vx(), mctrack.vy(), mctrack.vz(), mctrack.vt(), mcflags);
330  }
331 
332  fCounter = 0;
333  fNewLabels.clear();
334  fNewLabelsReversed.clear();
335  fMCFlags.clear();
336  fEventIdx.clear();
337  }
