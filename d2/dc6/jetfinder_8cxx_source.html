100  template <typename T>
101  bool processInit(T const& collision)
102  {
103  if (!collision.alias()[kINT7]) {
104  return false; //remove hard code
105  }
106  if (!collision.sel7()) {
107  return false; //remove hard code
108  }
109 
110  jets.clear();
111  inputParticles.clear();
112 
113  return true;
114  }

116  template <typename T>
117  void processImpl(T const& collision)
118  {
119  auto jetRValues = static_cast<std::vector<double>>(jetR);
120  for (auto R : jetRValues) {
121  // Update jet finder R and find jets
122  jetFinder.jetR = R;
123  fastjet::ClusterSequenceArea clusterSeq(jetFinder.findJets(inputParticles, jets));
124 
125  for (const auto& jet : jets) {
126  jetsTable(collision, jet.pt(), jet.eta(), jet.phi(),
127  jet.E(), jet.m(), jet.area(), std::round(R * 100));
128  hJetPt->Fill(jet.pt(), R);
129  hJetPhi->Fill(jet.phi(), R);
130  hJetEta->Fill(jet.eta(), R);
131  hJetN->Fill(jet.constituents().size(), R);
132  for (const auto& constituent : jet.constituents()) { //event or jetwise
133  if (DoConstSub) {
134  // Since we're copying the consituents, we can combine the tracks and clusters together
135  // We only have to keep the uncopied versions separated due to technical constraints.
136  constituentsSubTable(jetsTable.lastIndex(), constituent.pt(), constituent.eta(), constituent.phi(),
137  constituent.E(), constituent.m(), constituent.user_index());
138  }
139  if (constituent.user_index() < 0) {
140  // Cluster
141  // -1 to account for the convention of negative indices for clusters.
142  clusterConstituentsTable(jetsTable.lastIndex(), -1 * constituent.user_index());
143  } else {
144  // Tracks
145  trackConstituentsTable(jetsTable.lastIndex(), constituent.user_index());
146  }
147  }
148  }
149  }
150  }

152  void processParticleLevel(aod::McCollision const& collision, aod::McParticles const& particles)
153  {
154  // Setup
155  // As of June 2021, I don't think enums are supported as configurables, so we have to handle the conversion here.
156  // TODO: Double cast is to work around conversion failure.
157  auto _jetType = static_cast<JetType_t>(static_cast<int>(jetType));
158 
159  // Initialziation and event selection
160  // TODO: MC event selection?
161  jets.clear();
162  inputParticles.clear();
163 
164  // As of June 2021, how best to check for charged particles? It doesn't seem to be in
165  // the McParticles table, so for now we select by PID.
166  // charged hadron (e-, mu-, pi+, K+, p+, Sigma+, Sigma-, Xi-, Omega-)
167  // We define full jets as taking everything, and neutral jets as not charged.
168  std::vector<unsigned int> selectedPIDs{11, 13, 211, 321, 2212, 3222, 3112, 3312, 3334};
169  auto selectedBegin = selectedPIDs.begin();
170  auto selectedEnd = selectedPIDs.end();
171  for (auto& particle : particles) {
172  if (_jetType != JetType_t::full) {
173  bool foundChargedParticle = (std::find(selectedBegin, selectedEnd, particle.pdgCode()) == selectedEnd);
174  if (_jetType == JetType_t::charged && foundChargedParticle == false) {
175  continue;
176  }
177  if (_jetType == JetType_t::neutral && foundChargedParticle == true) {
178  continue;
179  }
180  }
181 
182  inputParticles.emplace_back(
183  fastjet::PseudoJet(
184  particle.px(), particle.py(), particle.pz(), particle.e()));
185  inputParticles.back().set_user_index(particle.globalIndex());
186  }
187 
188  processImpl(collision);
189  }

193  void processData(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision,
194  soa::Filtered<soa::Join<aod::Tracks, aod::TrackSelection>> const& tracks,
195  aod::EMCALClusters const& clusters)
196  {
197  // Setup
198  // As of June 2021, I don't think enums are supported as configurables, so we have to handle the conversion here.
199  // FIXME: Double cast is to work around conversion failure.
200  auto _jetType = static_cast<JetType_t>(static_cast<int>(jetType));
201 
202  // Initialziation and event selection
203  bool accepted = processInit(collision);
204  if (!accepted) {
205  return;
206  }
207 
208  if (_jetType == JetType_t::full || _jetType == JetType_t::charged) {
209  for (auto& track : tracks) {
210  fillConstituents(track, inputParticles);
211  inputParticles.back().set_user_index(track.globalIndex());
212  }
213  }
214  if (_jetType == JetType_t::full || _jetType == JetType_t::neutral) {
215  for (auto& cluster : clusters) {
216  // The right thing to do here would be to fully calculate the momentum correcting for the vertex position.
217  // However, it's not clear that this functionality exists yet (21 June 2021)
218  double pt = cluster.energy() / std::cosh(cluster.eta());
219  inputParticles.emplace_back(
220  fastjet::PseudoJet(
221  pt * std::cos(cluster.phi()),
222  pt * std::sin(cluster.phi()),
223  pt * std::sinh(cluster.eta()),
224  cluster.energy()));
225  // Clusters are denoted with negative indices.
226  inputParticles.back().set_user_index(-1 * cluster.globalIndex());
227  }
228  }
229 
230  processImpl(collision);
231  }

233  PROCESS_SWITCH(JetFinderTask, processData, "Data jet finding", true);

248 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
249 {
250  auto jetInputData = cfgc.options().get<std::string>("jet-input-data");
251  const std::map<std::string, JetInputData_t> jetInputDataTypes = {
252  {"Data", JetInputData_t::Data},
253  {"MCParticleLevel", JetInputData_t::MCParticleLevel},
254  {"MCDetectorLevel", JetInputData_t::MCDetectorLevel},
255  {"HybridIntermediate", JetInputData_t::HybridIntermediate},
256  {"", JetInputData_t::Data}, // Default to data
257  };
258  // TODO: Is there a better way to do this?
259  // TODO: String validation and normalization. There's too much room for error with enumerations.
260  auto jetMatching = cfgc.options().get<std::string>("jet-input-data");
261  // Tokenize using stringstream
262  std::vector<std::string> matchingOptions;
263  std::stringstream ss;
264  ss << jetMatching;
265  while (ss.good()) {
266  std::string substring;
267  getline(ss, substring, ',');
268  matchingOptions.push_back(substring);
269  }
270  std::vector<o2::framework::DataProcessorSpec> tasks;
271  for (auto opt : matchingOptions) {
272  auto jetData = jetInputDataTypes.at(opt);
273  switch (jetData) {
274  case JetInputData_t::MCParticleLevel:
275  tasks.emplace_back(
276  adaptAnalysisTask<JetFinderMCParticleLevel>(cfgc,
277  SetDefaultProcesses{{{"processParticleLevel", true}, {"processData", false}}}, TaskName{"jet-finder-MC"}));
278  break;
279  case JetInputData_t::MCDetectorLevel:
280  tasks.emplace_back(
281  adaptAnalysisTask<JetFinderMCDetectorLevel>(cfgc, TaskName{"jet-finder-MC-detector-level"}));
282  break;
283  case JetInputData_t::HybridIntermediate:
284  tasks.emplace_back(
285  adaptAnalysisTask<JetFinderHybridIntermediate>(cfgc, TaskName{"jet-finder-hybrid-intermedaite"}));
286  break;
287  case JetInputData_t::Data: // intentionally fall through to the default which is outside of the switch.
288  default:
289  tasks.emplace_back(
290  adaptAnalysisTask<JetFinderData>(cfgc, TaskName{"jet-finder-data"}));
291  break;
292  }
293  }
294  return WorkflowSpec{tasks};
295 }
