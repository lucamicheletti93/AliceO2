89  template <typename T1, typename T2>
90  bool selectionTopol(const T1& candidate, const T2& trackPos, const T2& trackNeg, int& selEE, int& selMuMu)
91  {
92  auto candpT = candidate.pt();
93  auto pTBin = findBin(pTBins, candpT);
94  if (pTBin == -1) {
95  return false;
96  }
97 
98  // check that the candidate pT is within the analysis range
99  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
100  return false;
101  }
102 
103  // cut on e+ e− invariant mass
104  if (std::abs(InvMassJpsiToEE(candidate) - RecoDecay::getMassPDG(pdg::Code::kJpsi)) > cuts->get(pTBin, "m")) {
105  selEE = 0;
106  }
107 
108  // cut on μ+ μ− invariant mass
109  if (std::abs(InvMassJpsiToMuMu(candidate) - RecoDecay::getMassPDG(pdg::Code::kJpsi)) > cuts->get(pTBin, "m")) {
110  selMuMu = 0;
111  }
112 
113  if (selEE == 0 && selMuMu == 0) {
114  return false;
115  }
116 
117  // cut on daughter pT (same cut used for both channels)
118  if (trackNeg.pt() < cuts->get(pTBin, "pT El") || trackPos.pt() < cuts->get(pTBin, "pT El")) {
119  return false;
120  }
121 
122  // cut on daughter DCA - need to add secondary vertex constraint here
123  if (std::abs(trackNeg.dcaPrim0()) > cuts->get(pTBin, "DCA_xy") || std::abs(trackPos.dcaPrim0()) > cuts->get(pTBin, "DCA_xy")) {
124  return false;
125  }
126 
127  // cut on daughter DCA - need to add secondary vertex constraint here
128  if (std::abs(trackNeg.dcaPrim1()) > cuts->get(pTBin, "DCA_z") || std::abs(trackPos.dcaPrim1()) > cuts->get(pTBin, "DCA_z")) {
129  return false;
130  }
131 
132  // cut on chi2 point of closest approach
133  if (std::abs(candidate.chi2PCA()) > cuts->get(pTBin, "chi2PCA")) {
134  return false;
135  }
136  return true;
137  }

141  void processAlice2(aod::HfCandProng2 const& candidates, aod::BigTracksPID const&)
142  {
143  TrackSelectorPID selectorElectron(kElectron);
144  selectorElectron.setRangePtTPC(d_pidTPCMinpT, d_pidTPCMaxpT);
145  selectorElectron.setRangeNSigmaTPC(-d_nSigmaTPC, d_nSigmaTPC);
146  selectorElectron.setRangePtTOF(d_pidTOFMinpT, d_pidTOFMaxpT);
147  selectorElectron.setRangeNSigmaTOF(-d_nSigmaTOF, d_nSigmaTOF);
148  selectorElectron.setRangeNSigmaTOFCondTPC(-d_nSigmaTOFCombined, d_nSigmaTOFCombined);
149  selectorElectron.setRangePtRICH(d_pidRICHMinpT, d_pidRICHMaxpT);
150  selectorElectron.setRangeNSigmaRICH(-d_nSigmaRICH, d_nSigmaRICH);
151  selectorElectron.setRangeNSigmaRICHCondTOF(-d_nSigmaRICHCombinedTOF, d_nSigmaRICHCombinedTOF);
152 
153  // looping over 2-prong candidates
154  for (auto& candidate : candidates) {
155 
156  if (!(candidate.hfflag() & 1 << DecayType::JpsiToEE) && !(candidate.hfflag() & 1 << DecayType::JpsiToMuMu)) {
157  hfSelJpsiCandidate(0, 0, 0, 0, 0, 0, 0, 0, 0);
158  //hfSelJpsiCandidate(0, 0);
159  continue;
160  }
161 
162  auto trackPos = candidate.index0_as<aod::BigTracksPID>(); // positive daughter
163  auto trackNeg = candidate.index1_as<aod::BigTracksPID>(); // negative daughter
164 
165  int selectedEETopol = 1;
166  int selectedEETpc = 1;
167  int selectedEETof = 1;
168  int selectedEERich = 1;
169  int selectedEETofRich = 1;
170  int selectedMuMuTopol = 1;
171  int selectedMuMuMid = 1;
172  int selectedEE = 1;
173  int selectedMuMu = 1;
174 
175  // track selection level need to add special cuts (additional cuts on decay length and d0 norm)
176 
177  if (!selectionTopol(candidate, trackPos, trackNeg, selectedEETopol, selectedMuMuTopol)) {
178  selectedEETopol = 0;
179  selectedMuMuTopol = 0;
180  selectedEE = 0;
181  selectedMuMu = 0;
182  //if (!selectionTopol(candidate, trackPos, trackNeg, selectedEE, selectedMuMu)) {
183  //hfSelJpsiCandidate(0, 0);
184  //continue;
185  }
186 
187  // track-level electron PID TOF selection
188  if (selectorElectron.getStatusTrackPIDTOF(trackPos) == TrackSelectorPID::Status::PIDRejected ||
189  selectorElectron.getStatusTrackPIDTOF(trackNeg) == TrackSelectorPID::Status::PIDRejected) {
190  selectedEETof = 0;
191  selectedEE = 0;
192  //if (selectedMuMu == 0) {
193  // hfSelJpsiCandidate(0, 0);
194  // continue;
195  //}
196  }
197 
198  // track-level electron PID TPC selection
199  if (selectorElectron.getStatusTrackPIDTPC(trackPos) == TrackSelectorPID::Status::PIDRejected ||
200  selectorElectron.getStatusTrackPIDTPC(trackNeg) == TrackSelectorPID::Status::PIDRejected) {
201  selectedEETpc = 0;
202  selectedEE = 0;
203  }
204 
205  hfSelJpsiCandidate(selectedEE,
206  selectedMuMu,
207  selectedEETopol,
208  selectedEETpc,
209  selectedEETof,
210  selectedEERich,
211  selectedEETofRich,
212  selectedMuMuTopol,
213  selectedMuMuMid);
214  //hfSelJpsiCandidate(selectedEE, selectedMuMu);
215  }
216  }
