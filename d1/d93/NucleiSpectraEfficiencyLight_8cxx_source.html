74  // loop over generated particles and fill generated particles
75  //

92 
93  HistogramRegistry spectra{"spectraRec", {}, OutputObjHandlingPolicy::AnalysisObject, true, true};
94 
95  void init(o2::framework::InitContext&)
96  {
97  std::vector<double> ptBinning = {0.5, 0.6, 0.7, 0.8, 0.9, 1.0, 1.1, 1.2, 1.3, 1.4, 1.5, 1.6,
98  1.8, 2.0, 2.2, 2.4, 2.8, 3.2, 3.6, 4., 5., 6., 8., 10., 12., 14.};
99  //
100  AxisSpec ptAxis = {ptBinning, "#it{p}_{T} (GeV/#it{c})"};
101  //
102  spectra.add("histEvSel", "eventselection", HistType::kTH1D, {{10, -0.5, 9.5}});
103  spectra.add("histRecVtxZ", "collision z position", HistType::kTH1F, {{200, -20., +20., "z position (cm)"}});
104  spectra.add("histRecPt", "reconstructed particles", HistType::kTH1F, {ptAxis});
105  spectra.add("histTpcSignal", "Specific energy loss", HistType::kTH2F, {{600, -6., 6., "#it{p/z} (GeV/#it{c})"}, {1400, 0, 1400, "d#it{E} / d#it{X} (a. u.)"}});
106  spectra.add("histTofSignalData", "TOF signal", HistType::kTH2F, {{600, -6., 6., "#it{p} (GeV/#it{c})"}, {500, 0.0, 1.2, "#beta (TOF)"}});
107  spectra.add("histTpcNsigma", "n-sigma TPC", HistType::kTH2F, {ptAxis, {200, -100., +100., "n#sigma_{He} (a. u.)"}});
108  spectra.add("histItsClusters", "number of ITS clusters", HistType::kTH1F, {{10, -0.5, +9.5, "number of ITS clusters"}});
109  spectra.add("histDcaXYprimary", "dca XY primary particles", HistType::kTH1F, {{200, -1., +1., "dca XY (cm)"}});
110  spectra.add("histDcaXYsecondary", "dca XY secondary particles", HistType::kTH1F, {{200, -1., +1., "dca XY (cm)"}});
111  }
112 
113  Configurable<float> cfgCutVertex{"cfgCutVertex", 10.0f, "Accepted z-vertex range"};
114  Configurable<float> cfgCutEta{"cfgCutEta", 0.8f, "Eta range for tracks"};
115  Configurable<float> nsigmacutLow{"nsigmacutLow", -10.0, "Value of the Nsigma cut"};
116  Configurable<float> nsigmacutHigh{"nsigmacutHigh", +10.0, "Value of the Nsigma cut"};
117 
118  Filter collisionFilter = nabs(aod::collision::posZ) < cfgCutVertex;
119  //Filter trackFilter = (nabs(aod::track::eta) < cfgCutEta) && (aod::track::isGlobalTrack == (uint8_t) true);
120 
121  using TrackCandidates = soa::Join<aod::Tracks, aod::TracksExtra, aod::McTrackLabels, aod::pidTPCFullHe, aod::pidTOFFullHe>;
122 
123  void process(soa::Filtered<soa::Join<aod::Collisions, aod::McCollisionLabels, aod::EvSels>>::iterator const& collision,
124  TrackCandidates const& tracks, aod::McParticles& mcParticles, aod::McCollisions const& mcCollisions)
125  {
126  //
127  // check event selection
128  //
129  spectra.get<TH1>(HIST("histEvSel"))->Fill("all", 1.f);
130  if (collision.sel8()) {
131  spectra.get<TH1>(HIST("histEvSel"))->Fill("sel8", 1.f);
132  } else {
133  return;
134  }
135  //
136  // check the vertex-z distribution
137  //
138  spectra.fill(HIST("histRecVtxZ"), collision.posZ());
139  //
140  // loop over reconstructed particles and fill reconstructed tracks
141  //
142  for (auto track : tracks) {
143  TLorentzVector lorentzVector{};
144  lorentzVector.SetPtEtaPhiM(track.pt() * 2.0, track.eta(), track.phi(), constants::physics::MassHelium3);
145  if (lorentzVector.Rapidity() < -0.5 || lorentzVector.Rapidity() > 0.5) {
146  continue;
147  }
148  //
149  // fill QA histograms
150  //
151  float nSigmaHe3 = track.tpcNSigmaHe();
152  nSigmaHe3 += 94.222101 * TMath::Exp(-0.905203 * track.tpcInnerParam());
153  //
154  // TPC-QA
155  //
156  if (track.itsNCls() > 0) {
157  spectra.fill(HIST("histTpcSignal"), track.tpcInnerParam() * track.sign(), track.tpcSignal());
158  }
159  spectra.fill(HIST("histTpcNsigma"), track.tpcInnerParam(), nSigmaHe3);
160  //
161  // ITS-QA
162  //
163  spectra.fill(HIST("histItsClusters"), track.itsNCls());
164  //
165  // TOF-QA
166  //
167  if (track.hasTOF()) {
168  Float_t tofTime = track.tofSignal();
169  Float_t tofLength = track.length();
170  Float_t beta = tofLength / (TMath::C() * 1e-10 * tofTime);
171  spectra.fill(HIST("histTofSignalData"), track.tpcInnerParam() * track.sign(), beta);
172  }
173  //
174  // dca to primary vertex -- wait for tracksextented
175  //
176  //spectra.fill(HIST("histDcaXYprimary"), track.dcaXY());
177  //
178  // fill histograms
179  //
180  if (nSigmaHe3 > nsigmacutLow && nSigmaHe3 < nsigmacutHigh) {
181  // check on perfect PID
182  if (track.mcParticle().pdgCode() != -1000020030) {
183  continue;
184  }
185  // fill reconstructed histogram
186  spectra.fill(HIST("histRecPt"), track.pt() * 2.0);
187  }
188  }
189  }
190 };

206  if (rec) {

209  //
