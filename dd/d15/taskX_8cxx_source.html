152  void process(soa::Filtered<soa::Join<aod::HfCandX, aod::HFSelXToJpsiPiPiCandidate, aod::HfCandXMCRec>> const& candidates,
153  soa::Join<aod::McParticles, aod::HfCandXMCGen> const& particlesMC, aod::BigTracksMC const& tracks)
154  {
155  // MC rec.
156  //Printf("MC Candidates: %d", candidates.size());
157  for (auto& candidate : candidates) {
158  if (!(candidate.hfflag() & 1 << XToJpsiPiPi)) {
159  continue;
160  }
161  if (cutYCandMax >= 0. && std::abs(YX(candidate)) > cutYCandMax) {
162  continue;
163  }
164  if (candidate.flagMCMatchRec() == 1 << XToJpsiPiPi) {
165  auto indexMother = RecoDecay::getMother(particlesMC, candidate.index1_as<aod::BigTracksMC>().mcParticle_as<soa::Join<aod::McParticles, aod::HfCandXMCGen>>(), 9920443, true);
166  auto particleMother = particlesMC.iteratorAt(indexMother);
167  registry.fill(HIST("hPtGenSig"), particleMother.pt());
168  registry.fill(HIST("hPtRecSig"), candidate.pt());
169  registry.fill(HIST("hCPARecSig"), candidate.cpa(), candidate.pt());
170  registry.fill(HIST("hEtaRecSig"), candidate.eta(), candidate.pt());
171 
172  registry.fill(HIST("hDeclengthRecSig"), candidate.decayLength(), candidate.pt());
173  registry.fill(HIST("hMassRecSig"), InvMassXToJpsiPiPi(candidate), candidate.pt());
174  registry.fill(HIST("hd0Prong0RecSig"), candidate.impactParameter0(), candidate.pt());
175  registry.fill(HIST("hd0Prong1RecSig"), candidate.impactParameter1(), candidate.pt());
176  registry.fill(HIST("hd0Prong2RecSig"), candidate.impactParameter2(), candidate.pt());
177  registry.fill(HIST("hPtProng0RecSig"), candidate.ptProng0(), candidate.pt());
178  registry.fill(HIST("hPtProng1RecSig"), candidate.ptProng1(), candidate.pt());
179  registry.fill(HIST("hPtProng2RecSig"), candidate.ptProng2(), candidate.pt());
180  registry.fill(HIST("hChi2PCASig"), candidate.chi2PCA(), candidate.pt());
181  registry.fill(HIST("hCtSig"), CtX(candidate), candidate.pt());
182  registry.fill(HIST("hYSig"), YX(candidate), candidate.pt());
183  } else {
184  registry.fill(HIST("hPtRecBg"), candidate.pt());
185  registry.fill(HIST("hCPARecBg"), candidate.cpa(), candidate.pt());
186  registry.fill(HIST("hEtaRecBg"), candidate.eta(), candidate.pt());
187 
188  registry.fill(HIST("hDeclengthRecBg"), candidate.decayLength(), candidate.pt());
189  registry.fill(HIST("hMassRecBg"), InvMassXToJpsiPiPi(candidate), candidate.pt());
190  registry.fill(HIST("hd0Prong0RecBg"), candidate.impactParameter0(), candidate.pt());
191  registry.fill(HIST("hd0Prong1RecBg"), candidate.impactParameter1(), candidate.pt());
192  registry.fill(HIST("hd0Prong2RecBg"), candidate.impactParameter2(), candidate.pt());
193  registry.fill(HIST("hPtProng0RecBg"), candidate.ptProng0(), candidate.pt());
194  registry.fill(HIST("hPtProng1RecBg"), candidate.ptProng1(), candidate.pt());
195  registry.fill(HIST("hPtProng2RecBg"), candidate.ptProng2(), candidate.pt());
196  registry.fill(HIST("hChi2PCABg"), candidate.chi2PCA(), candidate.pt());
197  registry.fill(HIST("hCtBg"), CtX(candidate), candidate.pt());
198  registry.fill(HIST("hYBg"), YX(candidate), candidate.pt());
199  }
200  } // rec
201  // MC gen.
202  //Printf("MC Particles: %d", particlesMC.size());
203  for (auto& particle : particlesMC) {
204  if (particle.flagMCMatchGen() == 1 << XToJpsiPiPi) {
205  // TODO: add X(3872) mass such that we can use the getMassPDG function instead of hardcoded mass
206  if (cutYCandMax >= 0. && std::abs(RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, 3.87168)) > cutYCandMax) {
207  // Printf("MC Gen.: Y rejection: %g", RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, 3.87168));
208  continue;
209  }
210  registry.fill(HIST("hPtGen"), particle.pt());
211  registry.fill(HIST("hEtaGen"), particle.eta(), particle.pt());
212 
213  // properties of gen matched X(3872), to get a first look at some cuts
214  float ptProngs[3];
215  int counter = 0;
216  for (int iD = particle.daughter0Id(); iD <= particle.daughter1Id(); ++iD) {
217  ptProngs[counter] = particlesMC.iteratorAt(iD).pt();
218  counter++;
219  }
220  registry.fill(HIST("hPtGenProng0"), ptProngs[0], particle.pt());
221  registry.fill(HIST("hPtGenProng1"), ptProngs[1], particle.pt());
222  registry.fill(HIST("hPtGenProng2"), ptProngs[2], particle.pt());
223  }
224  } // gen
225  } // process
