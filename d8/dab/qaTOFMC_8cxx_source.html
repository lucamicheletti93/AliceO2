141  template <uint8_t pidIndex, typename T, typename TT>
142  void fillNsigma(const T& track, const TT& mcParticles, const float& nsigma)
143  {
144  const auto particle = track.mcParticle();
145  if (abs(particle.pdgCode()) == PDGs[pidIndex]) {
146 
147  histos.fill(HIST(hnsigmaMC[pidIndex]), track.pt(), nsigma);
148  // Selecting primaries
149  if (MC::isPhysicalPrimary(particle)) {
150  histos.fill(HIST(hnsigmaMCprm[pidIndex]), track.pt(), nsigma);
151  } else {
152  histos.fill(HIST(hnsigmaMCsec[pidIndex]), track.pt(), nsigma);
153  }
154  }
155  }

157  void process(aod::Collision const& collision,
158  soa::Join<aod::Tracks, aod::TracksExtra,
159  aod::pidTOFFullEl, aod::pidTOFFullMu, aod::pidTOFFullPi,
160  aod::pidTOFFullKa, aod::pidTOFFullPr, aod::pidTOFFullDe,
161  aod::pidTOFFullTr, aod::pidTOFFullHe, aod::pidTOFFullAl,
162  aod::McTrackLabels, aod::pidTOFbeta> const& tracks,
163  aod::McParticles& mcParticles)
164  {
165  if (collision.numContrib() < nMinNumberOfContributors) {
166  return;
167  }
168  const float collisionTime_ps = collision.collisionTime() * 1000.f;
169  unsigned int nTracksWithTOF = 0;
170  for (auto t : tracks) {
171  //
172  if (!t.hasTOF()) { // Skipping tracks without TOF
173  continue;
174  }
175  if (t.eta() < minEta || t.eta() > maxEta) {
176  continue;
177  }
178  nTracksWithTOF++;
179  float nsigma = -999.f;
180  if constexpr (pid_type == 0) {
181  nsigma = t.tofNSigmaEl();
182  } else if constexpr (pid_type == 1) {
183  nsigma = t.tofNSigmaMu();
184  } else if constexpr (pid_type == 2) {
185  nsigma = t.tofNSigmaPi();
186  } else if constexpr (pid_type == 3) {
187  nsigma = t.tofNSigmaKa();
188  } else if constexpr (pid_type == 4) {
189  nsigma = t.tofNSigmaPr();
190  } else if constexpr (pid_type == 5) {
191  nsigma = t.tofNSigmaDe();
192  } else if constexpr (pid_type == 6) {
193  nsigma = t.tofNSigmaTr();
194  } else if constexpr (pid_type == 7) {
195  nsigma = t.tofNSigmaHe();
196  } else if constexpr (pid_type == 8) {
197  nsigma = t.tofNSigmaAl();
198  }
199 
200  // Fill for all
201  histos.fill(HIST(hnsigma[pid_type]), t.pt(), nsigma);
202  histos.fill(HIST("event/tofbeta"), t.p(), t.beta());
203  const auto particle = t.mcParticle();
204  if (MC::isPhysicalPrimary(particle)) { // Selecting primaries
205  histos.fill(HIST(hnsigmaprm[pid_type]), t.pt(), nsigma);
206  histos.fill(HIST("event/tofbetaPrm"), t.p(), t.beta());
207  } else {
208  histos.fill(HIST(hnsigmasec[pid_type]), t.pt(), nsigma);
209  histos.fill(HIST("event/tofbetaSec"), t.p(), t.beta());
210  }
211  if (abs(particle.pdgCode()) == PDGs[pid_type]) { // Checking the PDG code
212  histos.fill(HIST("event/tofbetaMC"), t.pt(), t.beta());
213  if (MC::isPhysicalPrimary(particle)) {
214  histos.fill(HIST("event/tofbetaMCPrm"), t.pt(), t.beta());
215  } else {
216  histos.fill(HIST("event/tofbetaMCSec"), t.pt(), t.beta());
217  }
218  }
219  // Fill with PDG codes
220  fillNsigma<0>(t, mcParticles, nsigma);
221  fillNsigma<1>(t, mcParticles, nsigma);
222  fillNsigma<2>(t, mcParticles, nsigma);
223  fillNsigma<3>(t, mcParticles, nsigma);
224  fillNsigma<4>(t, mcParticles, nsigma);
225  fillNsigma<5>(t, mcParticles, nsigma);
226  fillNsigma<6>(t, mcParticles, nsigma);
227  fillNsigma<7>(t, mcParticles, nsigma);
228  fillNsigma<8>(t, mcParticles, nsigma);
229  }
230  histos.fill(HIST("event/T0"), nTracksWithTOF, collisionTime_ps);
231  }
232 };

234 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
235 {
236  auto workflow = WorkflowSpec{};

244  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Pion>>(cfgc, TaskName{"pidTOF-qa-Pi"}));
245  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Kaon>>(cfgc, TaskName{"pidTOF-qa-Ka"}));
246  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Proton>>(cfgc, TaskName{"pidTOF-qa-Pr"}));
247  }
248  if (cfgc.options().get<int>("qa-nuclei")) {
249  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Deuteron>>(cfgc, TaskName{"pidTOF-qa-De"}));
250  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Triton>>(cfgc, TaskName{"pidTOF-qa-Tr"}));
251  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Helium3>>(cfgc, TaskName{"pidTOF-qa-He"}));
252  workflow.push_back(adaptAnalysisTask<pidTOFTaskQA<PID::Alpha>>(cfgc, TaskName{"pidTOF-qa-Al"}));
253  }
254  return workflow;
