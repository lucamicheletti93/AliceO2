33 struct pidTOFTaskBeta {
34  using Trks = soa::Join<aod::Tracks, aod::TracksExtra>;
35  using Coll = aod::Collision;
36  Produces<aod::pidTOFbeta> tablePIDBeta;
37  tof::Beta<Coll, Trks::iterator, PID::Electron> responseElectron;
38  Configurable<float> expreso{"tof-expreso", 80, "Expected resolution for the computation of the expected beta"};
39 
40  void init(o2::framework::InitContext&)
41  {
42  responseElectron.mExpectedResolution = expreso.value;
43  }
44 
45  void process(Coll const& collision, Trks const& tracks)
46  {
47  tablePIDBeta.reserve(tracks.size());
48  for (auto const& trk : tracks) {
49  tablePIDBeta(responseElectron.GetBeta(collision, trk),
50  responseElectron.GetExpectedSigma(collision, trk),
51  responseElectron.GetExpectedSignal(collision, trk),
52  responseElectron.GetExpectedSigma(collision, trk),
53  responseElectron.GetSeparation(collision, trk));
54  }
55  }
56 };

107  void fillParticleHistos(const T& t, const float tof, const float exp_diff, const float nsigma)
108  {
109  histos.fill(HIST(hexpected[i]), t.p(), tof - exp_diff);

115  {

121  return;
122  }

124  histos.fill(HIST("event/length"), track.length());
125  histos.fill(HIST("event/eta"), track.eta());
126  histos.fill(HIST("event/tofsignal"), track.p(), track.tofSignal());
127  histos.fill(HIST("event/pt"), track.pt());
128  histos.fill(HIST("event/p"), track.p());
