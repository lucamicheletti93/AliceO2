64  template <typename T1, typename T2, typename T3>
65  bool selectionTopol(const T1& hfCandX, const T2& hfCandJpsi, const T3& trackPos, const T3& trackNeg)
66  {
67  auto candpT = hfCandX.pt();
68  int pTBin = findBin(pTBins, candpT);
69  if (pTBin == -1) {
70  // Printf("X topol selection failed at getpTBin");
71  return false;
72  }
73 
74  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
75  // Printf("X topol selection failed at cand pT check");
76  return false; //check that the candidate pT is within the analysis range
77  }
78 
79  // TODO: replace hardcoded mass with "RecoDecay::getMassPDG(9920443)"
80  if (TMath::Abs(InvMassXToJpsiPiPi(hfCandX) - 3.87168) > cuts->get(pTBin, "m")) {
81  // Printf("X topol selection failed at mass diff check");
82  return false; //check that mass difference is within bounds
83  }
84 
85  if ((hfCandJpsi.pt() < cuts->get(pTBin, "pT Jpsi")) || (trackNeg.pt() < cuts->get(pTBin, "pT Pi")) || (trackPos.pt() < cuts->get(pTBin, "pT Pi"))) {
86  // Printf("X topol selection failed at daughter pT check");
87  return false; //cut on daughter pT
88  }
89 
90  if (hfCandX.cpa() < cuts->get(pTBin, "CPA")) {
91  return false; // CPA check
92  }
93 
94  if ((TMath::Abs(hfCandX.impactParameter0()) > cuts->get(pTBin, "d0 Jpsi")) ||
95  (TMath::Abs(hfCandX.impactParameter1()) > cuts->get(pTBin, "d0 Pi")) ||
96  (TMath::Abs(hfCandX.impactParameter2()) > cuts->get(pTBin, "d0 Pi"))) {
97  return false; // DCA check on daughters
98  }
99 
100  // add more cuts: d0 product? PCA?
101 
102  return true;
103  }

186  void process(aod::HfCandX const& hfCandXs, aod::HfCandProng2, aod::BigTracksPID const& tracks)
187  {
188  for (auto& hfCandX : hfCandXs) { //looping over X candidates
189  // note the difference between Jpsi (index0) and pions (index1,2)
190  auto candJpsi = hfCandX.index0();
191  auto trackPos = hfCandX.index1_as<aod::BigTracksPID>(); //positive daughter
192  auto trackNeg = hfCandX.index2_as<aod::BigTracksPID>(); //negative daughter
193 
194  // check if flagged as X --> Jpsi Pi Pi
195  if (!(hfCandX.hfflag() & 1 << XToJpsiPiPi)) {
196  hfSelXToJpsiPiPiCandidate(0);
197  // Printf("X candidate selection failed at hfflag check");
198  continue;
199  }
200 
201  // daughter track validity selection
202  if (!daughterSelection(trackPos) || !daughterSelection(trackNeg)) {
203  hfSelXToJpsiPiPiCandidate(0);
204  // Printf("X candidate selection failed at daughter selection");
205  continue;
206  }
207 
208  //implement filter bit 4 cut - should be done before this task at the track selection level
209  //need to add special cuts (additional cuts on decay length and d0 norm)
210 
211  if (!selectionTopol(hfCandX, candJpsi, trackPos, trackNeg)) {
212  hfSelXToJpsiPiPiCandidate(0);
213  // Printf("X candidate selection failed at selection topology");
214  continue;
215  }
216 
217  if (selectionPID(trackPos) == 0 || selectionPID(trackNeg) == 0) {
218  hfSelXToJpsiPiPiCandidate(0);
219  // Printf("X candidate selection failed at selection PID");
220  continue;
221  }
222 
223  hfSelXToJpsiPiPiCandidate(1);
224  // Printf("X candidate selection successful, candidate should be selected");
225  }
226  }
