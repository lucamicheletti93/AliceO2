72  void process(soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelJpsiCandidate>> const& candidates)
73  {
74  int decayMode = d_modeJpsiToMuMu ? DecayType::JpsiToMuMu : DecayType::JpsiToEE;
75 
76  for (auto& candidate : candidates) {
77 
78  if (!(candidate.hfflag() & 1 << decayMode)) {
79  continue;
80  }
81  if (d_selectionFlagJpsi > 0) {
82  if (d_modeJpsiToMuMu) {
83  if (candidate.isSelJpsiToMuMuTopol() <= 0) {
84  continue;
85  }
86  } else {
87  if (candidate.isSelJpsiToEETopol() <= 0) {
88  continue;
89  }
90  }
91  }
92  if (cutYCandMax >= 0. && std::abs(YJpsi(candidate)) > cutYCandMax) {
93  continue;
94  }
95 
96  if (d_modeJpsiToMuMu) {
97  registry.fill(HIST("hmass"), InvMassJpsiToMuMu(candidate), candidate.pt());
98  } else {
99  registry.fill(HIST("hmass"), InvMassJpsiToEE(candidate), candidate.pt());
100  }
101  registry.fill(HIST("hptcand"), candidate.pt());
102  registry.fill(HIST("hptprong0"), candidate.ptProng0());
103  registry.fill(HIST("hptprong1"), candidate.ptProng1());
104  registry.fill(HIST("hdeclength"), candidate.decayLength(), candidate.pt());
105  registry.fill(HIST("hdeclengthxy"), candidate.decayLengthXY(), candidate.pt());
106  registry.fill(HIST("hd0Prong0"), candidate.impactParameter0(), candidate.pt());
107  registry.fill(HIST("hd0Prong1"), candidate.impactParameter1(), candidate.pt());
108  registry.fill(HIST("hd0d0"), candidate.impactParameterProduct(), candidate.pt());
109  registry.fill(HIST("hCPA"), candidate.cpa(), candidate.pt());
110  registry.fill(HIST("hEta"), candidate.eta(), candidate.pt());
111  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter0(), candidate.pt());
112  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter1(), candidate.pt());
113  registry.fill(HIST("hDecLenErr"), candidate.errorDecayLength(), candidate.pt());
114  registry.fill(HIST("hDecLenXYErr"), candidate.errorDecayLengthXY(), candidate.pt());
115  }
116  }
